#sparql PREFIX datacite: <http://purl.org/spar/datacite/>
PREFIX dcterms: <http://purl.org/dc/terms/>
PREFIX literal: <http://www.essepuntato.it/2010/06/literalreification/>
PREFIX frbr: <http://purl.org/vocab/frbr/core#>
PREFIX fabio: <http://purl.org/spar/fabio/>

SELECT ?local_identifier ?name ?identifier_scheme ?identifier_value ?type
WHERE {
  BIND(<[[local_identifier]]> AS ?local_identifier)
  ?br frbr:partOf+ ?local_identifier .
  FILTER NOT EXISTS { ?local_identifier frbr:partOf ?parent. }
  ?local_identifier a ?type_uri .
  FILTER(?type_uri != fabio:Expression)
  BIND(STRAFTER(STR(?type_uri), "/spar/fabio/") AS ?oc_type)
  OPTIONAL {
    ?local_identifier dcterms:title ?name .
  }
  OPTIONAL {
    ?local_identifier datacite:hasIdentifier ?identifier .
    ?identifier datacite:usesIdentifierScheme ?identifier_scheme_uri ;
                literal:hasLiteralValue ?identifier_value .
    BIND(STRAFTER(STR(?identifier_scheme_uri), "http://purl.org/spar/datacite/") AS ?identifier_scheme)
  }
  BIND(
  IF(!BOUND(?oc_type), "unknown",
    IF(REGEX(?oc_type, "AcademicProceedings"), "conference",
      IF(REGEX(?oc_type, "BookSeries"), "other",
        IF(REGEX(?oc_type, "Book"), "book",
          IF(REGEX(?oc_type, "Journal"), "journal", "other"))))
  ) AS ?type)
} 

#url /venues
#type operation
#method get
#description Returns a list of venues matching the given filters when implemented.
#custom_params filter,skgif_filters.venues.yaml,Search filter.
Format: comma-separated `filter_name:filter_value` pairs (AND logic).
Filters in the first group use exact matching. For `cf.search.*`, matching is text contains.
**Attribute filters:**

- `acronym`
- `type`
- `identifiers.scheme`
- `identifiers.value`
- `name`

**Convenience filters (`cf.*`):**

- `cf.search.name`
