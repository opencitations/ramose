#format skg_if,to_skg_if,application/ld+json
#default_format skg_if
#sparql PREFIX datacite: <http://purl.org/spar/datacite/>
PREFIX literal: <http://www.essepuntato.it/2010/06/literalreification/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>

SELECT ?local_identifier ?name ?famliyName ?givenName ?identifier ?identifier_scheme ?identifier_value 
WHERE {
  BIND(<[[local_identifier]]> AS ?local_identifier)
  ?local_identifier a foaf:Agent.
  OPTIONAL { 
    ?local_identifier foaf:familyName ?familyName;
      foaf:givenName ?givenName .
  }
  OPTIONAL { ?local_identifier foaf:name ?name. }
  OPTIONAL {
    ?local_identifier datacite:hasIdentifier ?identifier .
    ?identifier datacite:usesIdentifierScheme ?identifier_scheme;
      literal:hasLiteralValue ?identifier_value.
  }
}

#url /persons
#type operation
#method get
#description Coming soon. Returns a list of persons matching the given filters when implemented.
#custom_params filter,skgif_filters.persons.yaml,Search filter.
Format: comma-separated `filter_name:filter_value` pairs (AND logic).
Filters in the first group use exact matching. For `cf.search.*`, matching is text contains.
**Attribute filters:**

- `identifiers.id`
- `identifiers.scheme`
- `given_name`
- `family_name`
- `name`
- `affiliations.affiliation.local_identifier` (currently empty)
- `affiliations.affiliation.name` (currently empty)
- `affiliations.affiliation.short_name` (currently empty)
- `affiliations.role` (currently empty)

**Convenience filters (`cf.*`):**

- `cf.search.family_name`
- `cf.search.given_name`
- `cf.search.name`
